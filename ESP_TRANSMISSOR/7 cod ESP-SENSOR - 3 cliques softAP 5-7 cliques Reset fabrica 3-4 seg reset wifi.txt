// - 3 cliques curtos + pausa de 1 segundo = SoftAP temporário
// - 5 a 7 cliques curtos + pausa de 1 segundo = Factory Reset
// - Pressionado entre 3 e 4 segundos = Reset Wi-Fi
// - Pressionado por 10 segundos não faz mais Factory Reset
// - LED pisca 2 vezes para SoftAP
// - LED pisca 3 vezes para Reset Wi-Fi
// - LED pisca 5 vezes para Factory Reset

// - GPIO9 permanece como botão BOOT.

#include "RMaker.h"
#include "WiFi.h"

// ==============================
// CONFIGURAÇÕES DO DISPOSITIVO
// ==============================
const char *service_name = "PROV_C3_01";
const char *pop = "12345678";
const char *service_key = "12345678";  // senha do SoftAP temporário

// ESP32-C3 SuperMini
static const int gpio_led = 8;      // LED onboard, normalmente invertido
static const int gpio_button = 9;   // botão BOOT, ativo em LOW

// ==============================
// ESTADO DO DISPOSITIVO
// ==============================
bool switch_state = true;

// Device único RainMaker
static Device *my_horta = NULL;

// NVS / Preferences
Preferences prefs;
const char *PREF_NS = "dasrum_rm";
const char *PREF_SOFTAP = "softap_once";

// ==============================
// DHT22 - TEMPERATURA E UMIDADE DO AR
// ==============================
static const int DHTPIN = 4;
#define DHTTYPE DHT22
DHT dht(DHTPIN, DHTTYPE);

// Compensação manual da temperatura
float valorParaSubtrair = 0.0f;

// ==============================
// SENSOR DE UMIDADE DO SOLO
// ==============================
static const int PINO_SENSOR_SOLO = 1;  // GPIO1 analógico
static const int valorSeco = 3370;
static const int valorMolhado = 1520;

// ==============================
// INTERVALOS DE LEITURA / ENVIO
// ==============================
static unsigned long last_sensor_read_ms = 0;
static const unsigned long SENSOR_SLOT_INTERVAL_MS = 4000; // 4 s por etapa

// Etapas:
// 0 = Umidade do Solo
// 1 = Temperatura do Ar
// 2 = Umidade do Ar
static uint8_t sensor_etapa = 0;

// ==============================
// CONTROLE DE VARIAÇÃO PARA PUBLICAÇÃO
// ==============================
static const float DELTA_UMIDADE_SOLO = 10.0f;   // publica se variar 10%
static const float DELTA_TEMPERATURA_AR = 0.2f;  // publica se variar 0,2 °C
static const float DELTA_UMIDADE_AR = 10.0f;     // publica se variar 10%

static float ultima_umidade_solo_publicada = NAN;
static float ultima_temperatura_ar_publicada = NAN;
static float ultima_umidade_ar_publicada = NAN;

// Cache da última leitura válida do DHT22
static bool ultima_leitura_dht_valida = false;
static float ultima_temperatura_dht = NAN;
static float ultima_umidade_ar_dht = NAN;

// ==============================
// MODOS DE PROVISIONAMENTO
// ==============================
enum ProvisionMode {
  PROV_MODE_BLE = 0,
  PROV_MODE_SOFTAP = 1
};

ProvisionMode currentProvMode = PROV_MODE_BLE;

// ==============================
// CONTROLE DO BOTÃO
// ==============================
static bool last_button_state = HIGH;
static unsigned long press_start_ms = 0;

// Controle dos cliques
static uint8_t click_count = 0;
static unsigned long sequence_start_ms = 0;
static unsigned long last_click_ms = 0;

// ==============================
// AJUSTES DE TEMPO DO BOTÃO
// ==============================
static const unsigned long DEBOUNCE_MS = 40;

// Clique curto
static const unsigned long SHORT_CLICK_MAX_MS = 1200;

// Tempo de pausa para decidir se a sequência terminou
static const unsigned long CLICK_DECISION_PAUSE_MS = 1000;

// Tempo máximo total da sequência de cliques
static const unsigned long CLICK_SEQUENCE_MAX_MS = 6000;

// Reset Wi-Fi por botão segurado
static const unsigned long WIFI_RESET_MIN_MS = 3000;
static const unsigned long WIFI_RESET_MAX_MS = 4000;

// Quantidade de cliques
static const uint8_t SOFTAP_CLICKS = 3;
static const uint8_t FACTORY_RESET_MIN_CLICKS = 5;
static const uint8_t FACTORY_RESET_MAX_CLICKS = 7;

// ==============================
// FUNÇÕES AUXILIARES
// ==============================
void setOnboardLed(bool on) {
  // LED invertido no ESP32-C3 SuperMini
  digitalWrite(gpio_led, on ? LOW : HIGH);
}

const char *getProvModeName() {
  return (currentProvMode == PROV_MODE_SOFTAP) ? "SoftAP" : "BLE";
}

const char *getQrTransportName() {
  return (currentProvMode == PROV_MODE_SOFTAP) ? "softap" : "ble";
}

void saveSoftAPOnce(bool enable) {
  prefs.begin(PREF_NS, false);
  prefs.putBool(PREF_SOFTAP, enable);
  prefs.end();
}

bool loadSoftAPOnce() {
  prefs.begin(PREF_NS, true);
  bool value = prefs.getBool(PREF_SOFTAP, false);
  prefs.end();
  return value;
}

void clearClickSequence() {
  click_count = 0;
  sequence_start_ms = 0;
  last_click_ms = 0;
}

void blinkFeedback(uint8_t times, unsigned long onMs, unsigned long offMs) {
  bool previous_state = switch_state;

  for (uint8_t i = 0; i < times; i++) {
    setOnboardLed(true);
    delay(onMs);
    setOnboardLed(false);
    delay(offMs);
  }

  setOnboardLed(previous_state);
}

void rebootToSoftAPRecovery(uint8_t totalClicks) {
  Serial.printf("Sequencia valida: %u cliques detectados.\n", totalClicks);
  Serial.println("Entrando em modo temporario SoftAP...");

  // Feedback solicitado: SoftAP = 2 piscadas
  blinkFeedback(2, 150, 150);

  saveSoftAPOnce(true);
  delay(300);
  ESP.restart();
}

void executeWiFiReset() {
  Serial.println("Botao: Reset Wi-Fi.");
  Serial.println("Apagando somente as credenciais de Wi-Fi salvas.");

  // Feedback solicitado: Reset Wi-Fi = 3 piscadas
  blinkFeedback(3, 150, 150);

  clearClickSequence();
  delay(300);
  RMakerWiFiReset(2);
}

void executeFactoryResetByClicks(uint8_t totalClicks) {
  Serial.printf("Sequencia valida: %u cliques detectados.\n", totalClicks);
  Serial.println("Botao: Factory Reset.");
  Serial.println("Apagando configuracoes do RainMaker/provisionamento. Firmware permanece gravado.");

  // Feedback solicitado: Factory Reset = 5 piscadas
  blinkFeedback(5, 150, 150);

  clearClickSequence();
  delay(300);
  RMakerFactoryReset(2);
}

void registerShortClick(unsigned long now) {
  if (click_count == 0) {
    click_count = 1;
    sequence_start_ms = now;
    last_click_ms = now;
  } else {
    if ((now - sequence_start_ms) > CLICK_SEQUENCE_MAX_MS) {
      Serial.println("Tempo maximo da sequencia excedido. Reiniciando contagem de cliques.");
      click_count = 1;
      sequence_start_ms = now;
      last_click_ms = now;
    } else {
      click_count++;
      last_click_ms = now;
    }
  }

  Serial.printf("Clique curto %u detectado. Aguardando pausa de 1 segundo para decidir...\n", click_count);

  if (click_count > FACTORY_RESET_MAX_CLICKS) {
    Serial.println("Mais de 7 cliques detectados. Sequencia ignorada para evitar reset acidental.");
    clearClickSequence();
  }
}

void processClickSequenceIfReady(unsigned long now) {
  if (click_count == 0) {
    return;
  }

  if ((now - last_click_ms) < CLICK_DECISION_PAUSE_MS) {
    return;
  }

  uint8_t totalClicks = click_count;
  clearClickSequence();

  if (totalClicks == SOFTAP_CLICKS) {
    rebootToSoftAPRecovery(totalClicks);
    return;
  }

  if (totalClicks >= FACTORY_RESET_MIN_CLICKS && totalClicks <= FACTORY_RESET_MAX_CLICKS) {
    executeFactoryResetByClicks(totalClicks);
    return;
  }

  Serial.printf("Sequencia de %u cliques ignorada.\n", totalClicks);
  Serial.println("Use 3 cliques para SoftAP ou 5 a 7 cliques para Factory Reset.");
}

// ==============================
// DIAGNÓSTICO DE PUBLICAÇÃO
// ==============================
float diferencaAbsoluta(float valorAtual, float ultimoValorPublicado) {
  float diferenca = valorAtual - ultimoValorPublicado;

  if (diferenca < 0) {
    diferenca = -diferenca;
  }

  return diferenca;
}

bool devePublicarValor(const char *nomeParametro, float valorAtual, float ultimoValorPublicado, float limite) {
  Serial.print("Diagnostico ");
  Serial.print(nomeParametro);
  Serial.print(" -> Atual: ");
  Serial.print(valorAtual);

  if (isnan(ultimoValorPublicado)) {
    Serial.print(" | Ultimo publicado: N/A");
    Serial.print(" | Limite: ");
    Serial.print(limite);
    Serial.println(" | Vai publicar: SIM - primeira publicacao");
    return true;
  }

  float diferenca = diferencaAbsoluta(valorAtual, ultimoValorPublicado);

  Serial.print(" | Ultimo publicado: ");
  Serial.print(ultimoValorPublicado);
  Serial.print(" | Diferenca: ");
  Serial.print(diferenca);
  Serial.print(" | Limite: ");
  Serial.print(limite);

  if (diferenca >= limite) {
    Serial.println(" | Vai publicar: SIM");
    return true;
  }

  Serial.println(" | Vai publicar: NAO");
  return false;
}

// ==============================
// SENSOR DE SOLO
// ==============================
int lerMediaSensorSolo(int amostras) {
  long soma = 0;
  for (int i = 0; i < amostras; i++) {
    soma += analogRead(PINO_SENSOR_SOLO);
    delay(10);
  }
  return soma / amostras;
}

int calcularUmidadePercentual(int valorLido, int seco, int molhado) {
  int percentual = map(valorLido, seco, molhado, 0, 100);
  if (percentual < 0) percentual = 0;
  if (percentual > 100) percentual = 100;
  return percentual;
}

const char* faixaUmidadeBruta(int valorBruto) {
  if (valorBruto <= 3370 && valorBruto >= 2908) return "MUITO SECO";
  if (valorBruto <= 2907 && valorBruto >= 2445) return "SECO";
  if (valorBruto <= 2444 && valorBruto >= 1983) return "UMIDADE MEDIA";
  if (valorBruto <= 1982 && valorBruto >= 1520) return "MUITO UMIDO";

  if (valorBruto > 3370) return "ACIMA DA FAIXA CALIBRADA";
  if (valorBruto < 1520) return "ABAIXO DA FAIXA CALIBRADA";

  return "FAIXA INDEFINIDA";
}

// ==============================
// EVENTOS DO PROVISIONAMENTO
// ==============================
void sysProvEvent(arduino_event_t *sys_event) {
  switch (sys_event->event_id) {
    case ARDUINO_EVENT_PROV_START:
      if (currentProvMode == PROV_MODE_BLE) {
        Serial.printf("\nProvisioning Started with name \"%s\" and PoP \"%s\" on BLE\n", service_name, pop);
      } else {
        Serial.printf("\nProvisioning Started with name \"%s\", key \"%s\" and PoP \"%s\" on SoftAP\n",
                      service_name, service_key, pop);
      }

      Serial.println("Scan this QR code from the provisioning application for Provisioning.");
      WiFiProv.printQR(service_name, pop, getQrTransportName());

      Serial.println();
      Serial.println("=== DADOS DE RECUPERACAO ===");
      Serial.printf("Modo atual: %s\n", getProvModeName());
      Serial.printf("Nome do dispositivo: %s\n", service_name);
      Serial.printf("POP: %s\n", pop);
      if (currentProvMode == PROV_MODE_SOFTAP) {
        Serial.printf("Senha do SoftAP: %s\n", service_key);
      }
      Serial.println("============================");
      break;

    case ARDUINO_EVENT_PROV_INIT:
      Serial.printf("Inicializando provisionamento %s...\n", getProvModeName());
      WiFiProv.disableAutoStop(10000);
      break;

    case ARDUINO_EVENT_PROV_CRED_RECV:
      Serial.println("Recebendo credenciais de Wi-Fi...");
      break;

    case ARDUINO_EVENT_PROV_CRED_SUCCESS:
      Serial.println("Credenciais recebidas com sucesso.");
      WiFiProv.endProvision();

      if (currentProvMode == PROV_MODE_SOFTAP) {
        Serial.println("Provisionamento em SoftAP concluido.");
        Serial.println("No proximo boot o dispositivo voltara ao modo normal BLE.");
        saveSoftAPOnce(false);
      }
      break;

    case ARDUINO_EVENT_WIFI_STA_CONNECTED:
      Serial.println("Conectado ao Wi-Fi.");
      break;

    case ARDUINO_EVENT_WIFI_STA_GOT_IP:
      Serial.print("IP: ");
      Serial.println(WiFi.localIP());
      break;

    case ARDUINO_EVENT_WIFI_STA_DISCONNECTED:
      Serial.println("Wi-Fi desconectado.");
      break;

    default:
      break;
  }
}

// ==============================
// CALLBACK DO DEVICE NO APP
// ==============================
void write_callback(Device *device, Param *param, const param_val_t val,
                    void *priv_data, write_ctx_t *ctx) {
  const char *param_name = param->getParamName();

  if (strcmp(param_name, "Power") == 0) {
    switch_state = val.val.b;
    setOnboardLed(switch_state);

    Serial.printf("App enviou Power: %s\n", switch_state ? "LIGADO" : "DESLIGADO");

    param->updateAndReport(val);
  }
}

// ==============================
// ESCOLHA DO MODO DE BOOT
// ==============================
void selectProvisionModeAtBoot() {
  bool softap_once = loadSoftAPOnce();

  if (softap_once) {
    currentProvMode = PROV_MODE_SOFTAP;
    Serial.println("Modo temporario SoftAP carregado da memoria.");
  } else {
    currentProvMode = PROV_MODE_BLE;
    Serial.println("Modo normal BLE carregado.");
  }
}

// ==============================
// INICIA O PROVISIONAMENTO
// ==============================
void startProvisioning() {
  WiFi.onEvent(sysProvEvent);

  if (currentProvMode == PROV_MODE_BLE) {
    WiFiProv.initProvision(NETWORK_PROV_SCHEME_BLE,
                           NETWORK_PROV_SCHEME_HANDLER_FREE_BTDM);

    RMaker.start();

    WiFiProv.beginProvision(NETWORK_PROV_SCHEME_BLE,
                            NETWORK_PROV_SCHEME_HANDLER_FREE_BTDM,
                            NETWORK_PROV_SECURITY_1,
                            pop,
                            service_name);
  } else {
    WiFiProv.initProvision(NETWORK_PROV_SCHEME_SOFTAP,
                           NETWORK_PROV_SCHEME_HANDLER_NONE);

    RMaker.start();

    WiFiProv.beginProvision(NETWORK_PROV_SCHEME_SOFTAP,
                            NETWORK_PROV_SCHEME_HANDLER_NONE,
                            NETWORK_PROV_SECURITY_1,
                            pop,
                            service_name,
                            service_key);
  }
}

// ==============================
// BOTÃO BOOT
// ==============================
// Nova lógica:
// - Clique curto = entra na contagem
// - 3 cliques curtos + 1 s de pausa = SoftAP temporário
// - 5 a 7 cliques curtos + 1 s de pausa = Factory Reset
// - Pressionado entre 3 e 4 s = Reset Wi-Fi
// - Pressionado por 10 s não faz mais Factory Reset
void handleBootButton() {
  bool current_state = digitalRead(gpio_button);
  unsigned long now = millis();

  if (last_button_state == HIGH && current_state == LOW) {
    press_start_ms = now;
  }

  if (last_button_state == LOW && current_state == HIGH) {
    unsigned long press_time = now - press_start_ms;

    if (press_time < DEBOUNCE_MS) {
      last_button_state = current_state;
      return;
    }

    if (press_time >= WIFI_RESET_MIN_MS && press_time <= WIFI_RESET_MAX_MS) {
      executeWiFiReset();
    } else if (press_time <= SHORT_CLICK_MAX_MS) {
      registerShortClick(now);
    } else {
      Serial.print("Pressionamento ignorado. Tempo pressionado: ");
      Serial.print(press_time);
      Serial.println(" ms.");
      Serial.println("Use 3 a 4 segundos para Reset Wi-Fi ou cliques curtos para SoftAP/Factory Reset.");
      clearClickSequence();
    }
  }

  processClickSequenceIfReady(now);

  last_button_state = current_state;
}

// ==============================
// LEITURA E PUBLICAÇÃO DOS SENSORES
// ==============================
void handleSensors() {
  unsigned long now = millis();

  if ((now - last_sensor_read_ms) < SENSOR_SLOT_INTERVAL_MS) {
    return;
  }

  last_sensor_read_ms = now;

  // ==================================================
  // ETAPA 0 -> UMIDADE DO SOLO
  // ==================================================
  if (sensor_etapa == 0) {
    int valorBrutoSolo = lerMediaSensorSolo(10);
    int umidadeSolo = calcularUmidadePercentual(valorBrutoSolo, valorSeco, valorMolhado);
    const char* faixaSolo = faixaUmidadeBruta(valorBrutoSolo);

    Serial.println("Etapa 0/2 -> Umidade do Solo");

    Serial.print("Solo -> Valor bruto: ");
    Serial.print(valorBrutoSolo);
    Serial.print(" | Umidade estimada: ");
    Serial.print(umidadeSolo);
    Serial.print("% | Faixa: ");
    Serial.println(faixaSolo);

    if (my_horta) {
      if (devePublicarValor("Umidade do Solo", (float)umidadeSolo, ultima_umidade_solo_publicada, DELTA_UMIDADE_SOLO)) {
        my_horta->updateAndReportParam("Umidade do Solo", (float)umidadeSolo);
        ultima_umidade_solo_publicada = (float)umidadeSolo;

        Serial.println("Publicado: Umidade do Solo");
      } else {
        Serial.println("Nao publicado: Umidade do Solo");
      }
    }

    sensor_etapa = 1;
    Serial.println("--------------------------------");
    return;
  }

  // ==================================================
  // ETAPA 1 -> LÊ DHT22 E PUBLICA TEMPERATURA DO AR
  // ==================================================
  if (sensor_etapa == 1) {
    float umidadeAr = dht.readHumidity();
    float temperaturaBruta = dht.readTemperature();

    Serial.println("Etapa 1/2 -> DHT22 + Temperatura do Ar");

    if (isnan(umidadeAr) || isnan(temperaturaBruta)) {
      ultima_leitura_dht_valida = false;
      Serial.println("Falha ao ler o sensor DHT22! Verifique as conexoes.");
    } else {
      float temperaturaFinal = temperaturaBruta - valorParaSubtrair;

      ultima_temperatura_dht = temperaturaFinal;
      ultima_umidade_ar_dht = umidadeAr;
      ultima_leitura_dht_valida = true;

      Serial.print("Temp. Original: ");
      Serial.print(temperaturaBruta);
      Serial.print(" *C | Temp. Compensada: ");
      Serial.print(temperaturaFinal);
      Serial.print(" *C | Umidade do Ar: ");
      Serial.print(umidadeAr);
      Serial.println(" %");

      if (my_horta) {
        if (devePublicarValor("Temperatura do Ar", temperaturaFinal, ultima_temperatura_ar_publicada, DELTA_TEMPERATURA_AR)) {
          my_horta->updateAndReportParam("Temperatura", temperaturaFinal);
          ultima_temperatura_ar_publicada = temperaturaFinal;

          Serial.println("Publicado: Temperatura do Ar");
        } else {
          Serial.println("Nao publicado: Temperatura do Ar");
        }
      }
    }

    sensor_etapa = 2;
    Serial.println("--------------------------------");
    return;
  }

  // ==================================================
  // ETAPA 2 -> USA ÚLTIMA LEITURA DO DHT22 E PUBLICA UMIDADE DO AR
  // ==================================================
  if (sensor_etapa == 2) {
    Serial.println("Etapa 2/2 -> Umidade do Ar usando ultima leitura do DHT22");

    if (!ultima_leitura_dht_valida || isnan(ultima_umidade_ar_dht)) {
      Serial.println("Sem leitura valida recente do DHT22. Umidade do Ar nao publicada.");
    } else {
      Serial.print("Umidade do Ar em cache: ");
      Serial.print(ultima_umidade_ar_dht);
      Serial.println(" %");

      if (my_horta) {
        if (devePublicarValor("Umidade do Ar", ultima_umidade_ar_dht, ultima_umidade_ar_publicada, DELTA_UMIDADE_AR)) {
          my_horta->updateAndReportParam("Umidade do Ar", ultima_umidade_ar_dht);
          ultima_umidade_ar_publicada = ultima_umidade_ar_dht;

          Serial.println("Publicado: Umidade do Ar");
        } else {
          Serial.println("Nao publicado: Umidade do Ar");
        }
      }
    }

    sensor_etapa = 0;
    Serial.println("Ciclo completo dos sensores: Solo -> Temperatura -> Umidade do Ar");
    Serial.println("--------------------------------");
    return;
  }

  // Segurança caso sensor_etapa receba algum valor inesperado
  sensor_etapa = 0;
}

// ==============================
// SETUP
// ==============================
void setup() {
  Serial.begin(115200);
  delay(1500);

  Serial.println();
  Serial.println("Bootando ESP RainMaker...");

  pinMode(gpio_led, OUTPUT);
  setOnboardLed(switch_state);

  pinMode(gpio_button, INPUT_PULLUP);
  last_button_state = digitalRead(gpio_button);

  analogReadResolution(12);
  dht.begin();

  Serial.println("Sensor de umidade do solo no ESP32-C3");
  Serial.println("Faixas brutas:");
  Serial.println("3370 ate 2908 -> MUITO SECO");
  Serial.println("2907 ate 2445 -> SECO");
  Serial.println("2444 ate 1983 -> UMIDADE MEDIA");
  Serial.println("1982 ate 1520 -> MUITO UMIDO");
  Serial.println("--------------------------------");

  selectProvisionModeAtBoot();

  Node my_node = RMaker.initNode("ESP32-C3 Inteligente");

  my_horta = new Device("Horta Inteligente", NULL, NULL);
  if (!my_horta) {
    Serial.println("Erro ao criar device Horta Inteligente.");
    return;
  }

  my_horta->addPowerParam(switch_state, "Power");

  Param temperature_param("Temperatura", "esp.param.temperature", value(0.0f),
                          PROP_FLAG_READ | PROP_FLAG_TIME_SERIES);
  my_horta->addParam(temperature_param);

  Param air_humidity_param("Umidade do Ar", "esp.param.humidity", value(0.0f),
                           PROP_FLAG_READ | PROP_FLAG_TIME_SERIES);
  my_horta->addParam(air_humidity_param);

  Param soil_humidity_param("Umidade do Solo", "esp.param.humidity", value(0.0f),
                            PROP_FLAG_READ | PROP_FLAG_TIME_SERIES);
  soil_humidity_param.addBounds(value(0.0f), value(100.0f), value(1.0f));
  my_horta->addParam(soil_humidity_param);

  my_horta->assignPrimaryParam(my_horta->getParamByName("Power"));
  my_horta->addCb(write_callback);

  my_node.addDevice(*my_horta);

  startProvisioning();

  Serial.printf("RainMaker iniciado em modo %s.\n", getProvModeName());
  Serial.println("Modo normal: BLE + QR.");
  Serial.println("Recuperacao sem QR: 3 cliques curtos no BOOT + pausa de 1 segundo.");
  Serial.println("O ESP reinicia em SoftAP temporario.");
  Serial.println("Apos provisionar com sucesso, o proximo boot volta ao BLE normal.");
  Serial.println("Factory Reset: 5 a 7 cliques curtos no BOOT + pausa de 1 segundo.");
  Serial.println("Reset Wi-Fi: manter BOOT pressionado entre 3 e 4 segundos.");
  Serial.println("Pressionar BOOT por 10 segundos nao executa mais Factory Reset.");
  Serial.println("LED feedback: 2 piscadas = SoftAP | 3 piscadas = Reset Wi-Fi | 5 piscadas = Factory Reset.");
  Serial.println("DHT22 ativo no GPIO4.");
  Serial.println("Sensor de solo ativo no GPIO1.");
  Serial.println("Sensores em etapas a cada 4 segundos.");
  Serial.println("Ciclo completo em aproximadamente 12 segundos.");
  Serial.println("Ordem de envio: Solo -> Temperatura do Ar -> Umidade do Ar.");
  Serial.println("Publicacao somente quando houver variacao relevante.");
  Serial.println("Monitor Serial mostra atual, ultimo publicado, diferenca e limite.");
  Serial.println("Graficos habilitados para Temperatura, Umidade do Ar e Umidade do Solo.");
}

// ==============================
// LOOP
// ==============================
void loop() {
