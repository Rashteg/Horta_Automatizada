void setOnboardLed(bool on) {
  digitalWrite(gpio_led, on ? LOW : HIGH);
}

const char *getProvModeName() {
  return (currentProvMode == PROV_MODE_SOFTAP) ? "SoftAP" : "BLE";
}

const char *getQrTransportName() {
  return (currentProvMode == PROV_MODE_SOFTAP) ? "softap" : "ble";
}

void saveSoftAPOnce(bool enable) {
  prefs.begin(PREF_NS, false);
  prefs.putBool(PREF_SOFTAP, enable);
  prefs.end();
}

bool loadSoftAPOnce() {
  prefs.begin(PREF_NS, true);
  bool value = prefs.getBool(PREF_SOFTAP, false);
  prefs.end();
  return value;
}

void clearClickSequence() {
  click_count = 0;
  sequence_start_ms = 0;
  last_click_ms = 0;
}

void blinkFeedback(uint8_t times, unsigned long onMs, unsigned long offMs) {
  bool previous_state = switch_state;

  for (uint8_t i = 0; i < times; i++) {
    setOnboardLed(true);
    delay(onMs);
    setOnboardLed(false);
    delay(offMs);
  }

  setOnboardLed(previous_state);
}

// SAÍDA FINAL DE IRRIGAÇÃO
// ==============================
void atualizarSaidaIrrigacao(bool novo_estado, const char *origem) {
  bool mudou = (novo_estado != saida_irrigacao_state);

  saida_irrigacao_state = novo_estado;

  Serial.print("Saída Irrigação atualizada por ");
  Serial.print(origem);
  Serial.print(": ");
  Serial.println(saida_irrigacao_state ? "ON" : "OFF");

  Serial.print("Modo: ");
  Serial.print(modo_automatico ? "AUTOMÁTICO" : "MANUAL");
  Serial.print(" | Pedido Solo: ");
  Serial.print(pedido_irrigacao_solo ? "SIM" : "NÃO");
  Serial.print(" | Pedido Temperatura: ");
  Serial.print(pedido_irrigacao_temperatura ? "SIM" : "NÃO");
  Serial.print(" | Comando Manual: ");
  Serial.println(comando_manual ? "ON" : "OFF");

  if (my_horta) {
    my_horta->updateAndReportParam("Saída Irrigação", saida_irrigacao_state);
  }

  if (mudou) {
    Serial.println("Estado final da irrigação mudou.");
  }
}

void aplicarSaidaIrrigacaoFinal(const char *origem) {
  if (modo_automatico) {
    bool novo_estado = (pedido_irrigacao_solo || pedido_irrigacao_temperatura);
    atualizarSaidaIrrigacao(novo_estado, origem);
  } else {
    atualizarSaidaIrrigacao(comando_manual, origem);
  }
}

void corrigirConfiguracoesAtivas() {
  if (faixa_umidade_min < UMIDADE_MIN_LIMITE) faixa_umidade_min = UMIDADE_MIN_LIMITE;
  if (faixa_umidade_min > UMIDADE_MAX_LIMITE) faixa_umidade_min = UMIDADE_MAX_LIMITE;

  if (faixa_umidade_max < UMIDADE_MIN_LIMITE) faixa_umidade_max = UMIDADE_MIN_LIMITE;
  if (faixa_umidade_max > UMIDADE_MAX_LIMITE) faixa_umidade_max = UMIDADE_MAX_LIMITE;

  if (faixa_umidade_min > faixa_umidade_max) {
    faixa_umidade_max = faixa_umidade_min;
  }

  if (faixa_temp_min < TEMP_MIN_LIMITE) faixa_temp_min = TEMP_MIN_LIMITE;
  if (faixa_temp_min > TEMP_MAX_LIMITE) faixa_temp_min = TEMP_MAX_LIMITE;

  if (faixa_temp_max < TEMP_MIN_LIMITE) faixa_temp_max = TEMP_MIN_LIMITE;
  if (faixa_temp_max > TEMP_MAX_LIMITE) faixa_temp_max = TEMP_MAX_LIMITE;

  if (faixa_temp_min > faixa_temp_max) {
    faixa_temp_max = faixa_temp_min;
  }

  if (tempo_irrigacao_temp_min < TEMPO_IRRIGACAO_MIN_LIMITE) {
    tempo_irrigacao_temp_min = TEMPO_IRRIGACAO_MIN_LIMITE;
  }

  if (tempo_irrigacao_temp_min > TEMPO_IRRIGACAO_MAX_LIMITE) {
    tempo_irrigacao_temp_min = TEMPO_IRRIGACAO_MAX_LIMITE;
  }

  if (my_horta) {
    my_horta->updateAndReportParam("Solo - Faixa Mínima (%)", faixa_umidade_min);
    my_horta->updateAndReportParam("Solo - Faixa Máxima (%)", faixa_umidade_max);
    my_horta->updateAndReportParam("Temp - Faixa Mínima (°C)", faixa_temp_min);
    my_horta->updateAndReportParam("Temp - Faixa Máxima (°C)", faixa_temp_max);
    my_horta->updateAndReportParam("Temp - Tempo Irrigação (min)", tempo_irrigacao_temp_min);
  }
}

bool temperaturaDentroDaFaixa(float temperatura) {
  return (temperatura >= (float)faixa_temp_min && temperatura <= (float)faixa_temp_max);
}

  ultima_umidade_solo_lida = umidadeSolo;
  ultima_umidade_solo_lida_valida = true;

  Serial.print("Umidade do solo atual: ");

  Serial.println("%");

  Serial.print("Solo - Faixa Mínima (%): ");

  Serial.println("%");

  Serial.print("Solo - Faixa Máxima (%): ");

  Serial.println("%");

void avaliarRegraIrrigacaoTemperatura(float temperatura) {
  ultima_temperatura_lida = temperatura;
  ultima_temperatura_lida_valida = true;

  bool dentro_da_faixa = temperaturaDentroDaFaixa(temperatura);
  bool entrada_na_faixa = (dentro_da_faixa && !temp_condicao_anterior);

  Serial.println();
  Serial.println("Avaliando regra: Irrigação por Temperatura");
  Serial.print("Temperatura atual: ");
  Serial.print(temperatura);
  Serial.println(" °C");

  Serial.print("Temp - Faixa Mínima (°C): ");
  Serial.print(faixa_temp_min);
  Serial.println(" °C");

  Serial.print("Temp - Faixa Máxima (°C): ");
  Serial.print(faixa_temp_max);
  Serial.println(" °C");

  Serial.print("Temp - Irrigação da Faixa: ");
  Serial.println(irrigacao_da_faixa_temp ? "ON" : "OFF");

  Serial.print("Temp - Tempo Irrigação (min): ");
  Serial.println(tempo_irrigacao_temp_min);

  if (!irrigacao_da_faixa_temp) {
    pedido_irrigacao_temperatura = false;
    temp_ciclo_ativo = false;
    temp_ciclo_fim_ms = 0;
    temp_condicao_anterior = dentro_da_faixa;

    aplicarSaidaIrrigacaoFinal("temperatura desabilitada");
    Serial.println("--------------------------------");
    return;
  }

  if (tempo_irrigacao_temp_min == 0) {
    pedido_irrigacao_temperatura = dentro_da_faixa;
    temp_ciclo_ativo = false;
    temp_ciclo_fim_ms = 0;
    temp_condicao_anterior = dentro_da_faixa;

    aplicarSaidaIrrigacaoFinal("temperatura contínua");
    Serial.println("--------------------------------");
    return;
  }

  if (entrada_na_faixa && !temp_ciclo_ativo) {
    unsigned long duracao_ms = (unsigned long)tempo_irrigacao_temp_min * 60000UL;

    temp_ciclo_ativo = true;
    temp_ciclo_fim_ms = millis() + duracao_ms;
    pedido_irrigacao_temperatura = true;

    Serial.println("Temperatura entrou na faixa. Ciclo temporizado iniciado.");
  }

  temp_condicao_anterior = dentro_da_faixa;

  aplicarSaidaIrrigacaoFinal("regra de temperatura");
  Serial.println("--------------------------------");
}

void reavaliarRegrasPorConfiguracao() {
  Serial.println();
  Serial.println("Aplicando configurações após atraso...");

  corrigirConfiguracoesAtivas();

  if (!irrigacao_da_faixa_solo) {
    pedido_irrigacao_solo = false;
    Serial.println("Solo - Irrigação da Faixa OFF: pedido de solo forçado para OFF.");
  }

  if (!irrigacao_da_faixa_temp) {
    pedido_irrigacao_temperatura = false;
    temp_ciclo_ativo = false;
    temp_ciclo_fim_ms = 0;
    Serial.println("Temp - Irrigação da Faixa OFF: pedido de temperatura forçado para OFF.");
  }

  if (ultima_umidade_solo_lida_valida) {
    avaliarRegraIrrigacaoSolo(ultima_umidade_solo_lida);
  }

  if (ultima_temperatura_lida_valida && !isnan(ultima_temperatura_lida)) {
    avaliarRegraIrrigacaoTemperatura(ultima_temperatura_lida);
  }

  if (!ultima_umidade_solo_lida_valida && !ultima_temperatura_lida_valida) {
    aplicarSaidaIrrigacaoFinal("configuração aplicada sem leituras válidas");
  }
}

void handleAplicarConfiguracoesPendentes() {
  if (!config_apply_pending) {
    return;
  }

  unsigned long now = millis();

  if ((long)(now - config_apply_due_ms) >= 0) {
    config_apply_pending = false;
    reavaliarRegrasPorConfiguracao();
  }
}

// ==============================
// TIMER TEMPERATURA
// ==============================
void handleTimerTemperatura() {
  if (!temp_ciclo_ativo) {
    return;
  }

  unsigned long now = millis();

  if ((long)(now - temp_ciclo_fim_ms) >= 0) {
    temp_ciclo_ativo = false;
    pedido_irrigacao_temperatura = false;
    temp_ciclo_fim_ms = 0;

    Serial.println("Tempo de irrigação por temperatura finalizado.");
    aplicarSaidaIrrigacaoFinal("fim do tempo de temperatura");
  }
}

void rebootToSoftAPRecovery(uint8_t totalClicks) {
  Serial.printf("Sequencia válida: %u cliques detectados.\n", totalClicks);
  Serial.println("Entrando em modo temporário SoftAP...");

  blinkFeedback(2, 150, 150);

  saveSoftAPOnce(true);
  delay(300);

}

  blinkFeedback(3, 150, 150);

  clearClickSequence();

}

void executeFactoryResetByClicks(uint8_t totalClicks) {
  Serial.printf("Sequência válida: %u cliques detectados.\n", totalClicks);
  Serial.println("Botão: Factory Reset.");
  Serial.println("Apagando configurações do RainMaker/provisionamento. Firmware permanece gravado.");

  blinkFeedback(5, 150, 150);

  clearClickSequence();
  delay(300);
  RMakerFactoryReset(2);
}

void registerShortClick(unsigned long now) {
  if (click_count == 0) {
    click_count = 1;
    sequence_start_ms = now;
    last_click_ms = now;
  } else {
    if ((now - sequence_start_ms) > CLICK_SEQUENCE_MAX_MS) {
      Serial.println("Tempo máximo da sequência excedido. Reiniciando contagem de cliques.");
      click_count = 1;
      sequence_start_ms = now;
      last_click_ms = now;
    } else {
      click_count++;
      last_click_ms = now;
    }
  }

  Serial.printf("Clique curto %u detectado. Aguardando pausa de 1 segundo para decidir...\n", click_count);

  if (click_count > FACTORY_RESET_MAX_CLICKS) {
    Serial.println("Mais de 7 cliques detectados. Sequência ignorada para evitar reset acidental.");
    clearClickSequence();
  }
}

void processClickSequenceIfReady(unsigned long now) {
  if (click_count == 0) {
    return;
  }

  if ((now - last_click_ms) < CLICK_DECISION_PAUSE_MS) {
    return;
  }

  uint8_t totalClicks = click_count;
  clearClickSequence();

  if (totalClicks == SOFTAP_CLICKS) {
    rebootToSoftAPRecovery(totalClicks);
    return;
  }

  if (totalClicks >= FACTORY_RESET_MIN_CLICKS && totalClicks <= FACTORY_RESET_MAX_CLICKS) {
    executeFactoryResetByClicks(totalClicks);
    return;
  }

  Serial.printf("Sequência de %u cliques ignorada.\n", totalClicks);
  Serial.println("Use 3 cliques para SoftAP ou 5 a 7 cliques para Factory Reset.");
}

// DIAGNÓSTICO DE PUBLICAÇÃO
// ==============================
float diferencaAbsoluta(float valorAtual, float ultimoValorPublicado) {
  float diferenca = valorAtual - ultimoValorPublicado;

  if (diferenca < 0) {
    diferenca = -diferenca;
  }

  return diferenca;
}

bool devePublicarValor(const char *nomeParametro, float valorAtual, float ultimoValorPublicado, float limite) {
  Serial.print("Diagnóstico ");
  Serial.print(nomeParametro);
  Serial.print(" -> Atual: ");
  Serial.print(valorAtual);

  if (isnan(ultimoValorPublicado)) {
    Serial.print(" | Último publicado: N/A");
    Serial.print(" | Limite: ");
    Serial.print(limite);
    Serial.println(" | Vai publicar: SIM - primeira publicação");
    return true;
  }

  float diferenca = diferencaAbsoluta(valorAtual, ultimoValorPublicado);

  Serial.print(" | Último publicado: ");
  Serial.print(ultimoValorPublicado);
  Serial.print(" | Diferença: ");
  Serial.print(diferenca);
  Serial.print(" | Limite: ");
  Serial.print(limite);

  if (diferenca >= limite) {
    Serial.println(" | Vai publicar: SIM");
    return true;
  }

  Serial.println(" | Vai publicar: NÃO");
  return false;
}

int lerMediaSensorSolo(int amostras) {
  long soma = 0;

  for (int i = 0; i < amostras; i++) {
    soma += analogRead(PINO_SENSOR_SOLO);
    delay(10);
  }

  return soma / amostras;
}

int calcularUmidadePercentual(int valorLido, int seco, int molhado) {
  int percentual = map(valorLido, seco, molhado, 0, 100);

  if (percentual < 0) percentual = 0;
  if (percentual > 100) percentual = 100;

  return percentual;
}

const char* faixaUmidadeBruta(int valorBruto) {
  if (valorBruto <= 3370 && valorBruto >= 2908) return "MUITO SECO";
  if (valorBruto <= 2907 && valorBruto >= 2445) return "SECO";
  if (valorBruto <= 2444 && valorBruto >= 1983) return "UMIDADE MÉDIA";
  if (valorBruto <= 1982 && valorBruto >= 1520) return "MUITO ÚMIDO";

  if (valorBruto > 3370) return "ACIMA DA FAIXA CALIBRADA";
  if (valorBruto < 1520) return "ABAIXO DA FAIXA CALIBRADA";

  return "FAIXA INDEFINIDA";
}

// ==============================
// EVENTOS DO PROVISIONAMENTO
// ==============================
void sysProvEvent(arduino_event_t *sys_event) {
  switch (sys_event->event_id) {
    case ARDUINO_EVENT_PROV_START:
      if (currentProvMode == PROV_MODE_BLE) {
        Serial.printf("\nProvisioning Started with name \"%s\" and PoP \"%s\" on BLE\n", service_name, pop);
      } else {
        Serial.printf("\nProvisioning Started with name \"%s\", key \"%s\" and PoP \"%s\" on SoftAP\n",
                      service_name, service_key, pop);
      }

      Serial.println("Scan this QR code from the provisioning application for Provisioning.");
      WiFiProv.printQR(service_name, pop, getQrTransportName());

      Serial.println();
      Serial.println("=== DADOS DE RECUPERAÇÃO ===");
      Serial.printf("Modo atual: %s\n", getProvModeName());
      Serial.printf("Nome do dispositivo: %s\n", service_name);
      Serial.printf("POP: %s\n", pop);
      if (currentProvMode == PROV_MODE_SOFTAP) {
        Serial.printf("Senha do SoftAP: %s\n", service_key);
      }
      Serial.println("============================");
      break;

    case ARDUINO_EVENT_PROV_INIT:
      Serial.printf("Inicializando provisionamento %s...\n", getProvModeName());
      WiFiProv.disableAutoStop(10000);
      break;

    case ARDUINO_EVENT_PROV_CRED_RECV:
      Serial.println("Recebendo credenciais de Wi-Fi...");
      break;

    case ARDUINO_EVENT_PROV_CRED_SUCCESS:
      Serial.println("Credenciais recebidas com sucesso.");
      WiFiProv.endProvision();

      if (currentProvMode == PROV_MODE_SOFTAP) {
        Serial.println("Provisionamento em SoftAP concluído.");
        Serial.println("No próximo boot o dispositivo voltará ao modo normal BLE.");
        saveSoftAPOnce(false);
      }
      break;

    case ARDUINO_EVENT_WIFI_STA_CONNECTED:
      Serial.println("Conectado ao Wi-Fi.");
      break;

    case ARDUINO_EVENT_WIFI_STA_GOT_IP:
      Serial.print("IP: ");
      Serial.println(WiFi.localIP());
      break;

    case ARDUINO_EVENT_WIFI_STA_DISCONNECTED:
      Serial.println("Wi-Fi desconectado.");
      break;

    default:
      break;
  }
}

// ==============================
// ESCOLHA DO MODO DE BOOT
// ==============================
void selectProvisionModeAtBoot() {
  bool softap_once = loadSoftAPOnce();

  if (softap_once) {
    currentProvMode = PROV_MODE_SOFTAP;
    Serial.println("Modo temporário SoftAP carregado da memória.");
  } else {
    currentProvMode = PROV_MODE_BLE;
    Serial.println("Modo normal BLE carregado.");
  }
}

// ==============================
// INICIA O PROVISIONAMENTO
// ==============================
void startProvisioning() {
  WiFi.onEvent(sysProvEvent);

  if (currentProvMode == PROV_MODE_BLE) {
    WiFiProv.initProvision(NETWORK_PROV_SCHEME_BLE,
                           NETWORK_PROV_SCHEME_HANDLER_FREE_BTDM);

    RMaker.start();

    WiFiProv.beginProvision(NETWORK_PROV_SCHEME_BLE,
                            NETWORK_PROV_SCHEME_HANDLER_FREE_BTDM,
                            NETWORK_PROV_SECURITY_1,
                            pop,
                            service_name);
  } else {
    WiFiProv.initProvision(NETWORK_PROV_SCHEME_SOFTAP,
                           NETWORK_PROV_SCHEME_HANDLER_NONE);

    RMaker.start();

    WiFiProv.beginProvision(NETWORK_PROV_SCHEME_SOFTAP,
                            NETWORK_PROV_SCHEME_HANDLER_NONE,
                            NETWORK_PROV_SECURITY_1,
                            pop,
                            service_name,
                            service_key);
  }
}

// ==============================
// BOTÃO BOOT
// ==============================
void handleBootButton() {
  bool current_state = digitalRead(gpio_button);
  unsigned long now = millis();

  if (last_button_state == HIGH && current_state == LOW) {
    press_start_ms = now;
  }

  if (last_button_state == LOW && current_state == HIGH) {
    unsigned long press_time = now - press_start_ms;

    if (press_time < DEBOUNCE_MS) {
      last_button_state = current_state;
      return;
    }

    if (press_time >= WIFI_RESET_MIN_MS && press_time <= WIFI_RESET_MAX_MS) {
      executeWiFiReset();
    } else if (press_time <= SHORT_CLICK_MAX_MS) {
      registerShortClick(now);
    } else {
      Serial.print("Pressionamento ignorado. Tempo pressionado: ");
      Serial.print(press_time);
      Serial.println(" ms.");
      Serial.println("Use 3 a 4 segundos para Reset Wi-Fi ou cliques curtos para SoftAP/Factory Reset.");
      clearClickSequence();
    }
  }

  processClickSequenceIfReady(now);

  last_button_state = current_state;
}

// ==============================
// LEITURA E PUBLICAÇÃO DOS SENSORES
// ==============================
void handleSensors() {
  unsigned long now = millis();

  if ((now - last_sensor_read_ms) < SENSOR_SLOT_INTERVAL_MS) {
    return;
  }

  last_sensor_read_ms = now;

  // ==================================================
  // ETAPA 0 -> UMIDADE DO SOLO
  // ==================================================
  if (sensor_etapa == 0) {
    int valorBrutoSolo = lerMediaSensorSolo(10);
    int umidadeSolo = calcularUmidadePercentual(valorBrutoSolo, valorSeco, valorMolhado);
    const char* faixaSolo = faixaUmidadeBruta(valorBrutoSolo);

    Serial.println("Etapa 0/2 -> Umidade do Solo");

    Serial.print("Solo -> Valor bruto: ");
    Serial.print(valorBrutoSolo);
    Serial.print(" | Umidade estimada: ");
    Serial.print(umidadeSolo);
    Serial.print("% | Faixa: ");
    Serial.println(faixaSolo);

    if (my_horta) {
      if (devePublicarValor("Umidade do Solo", (float)umidadeSolo, ultima_umidade_solo_publicada, DELTA_UMIDADE_SOLO)) {
        my_horta->updateAndReportParam("Umidade do Solo", (float)umidadeSolo);
        ultima_umidade_solo_publicada = (float)umidadeSolo;

        Serial.println("Publicado: Umidade do Solo");
      } else {
        Serial.println("Não publicado: Umidade do Solo");
      }
    }

    avaliarRegraIrrigacaoSolo(umidadeSolo);

    sensor_etapa = 1;
    Serial.println("--------------------------------");
    return;
  }

  // ==================================================
  // ETAPA 1 -> TEMPERATURA DO AR
  // ==================================================
  if (sensor_etapa == 1) {
    float umidadeAr = dht.readHumidity();
    float temperaturaBruta = dht.readTemperature();

    Serial.println("Etapa 1/2 -> DHT22 + Temperatura do Ar");

    if (isnan(umidadeAr) || isnan(temperaturaBruta)) {
      ultima_leitura_dht_valida = false;
      Serial.println("Falha ao ler o sensor DHT22! Verifique as conexões.");
    } else {
      float temperaturaFinal = temperaturaBruta - valorParaSubtrair;

      ultima_temperatura_dht = temperaturaFinal;
      ultima_umidade_ar_dht = umidadeAr;
      ultima_leitura_dht_valida = true;

      Serial.print("Temp. Original: ");
      Serial.print(temperaturaBruta);
      Serial.print(" *C | Temp. Compensada: ");
      Serial.print(temperaturaFinal);
      Serial.print(" *C | Umidade do Ar: ");
      Serial.print(umidadeAr);
      Serial.println(" %");

      if (my_horta) {
        if (devePublicarValor("Temperatura do Ar", temperaturaFinal, ultima_temperatura_ar_publicada, DELTA_TEMPERATURA_AR)) {
          my_horta->updateAndReportParam("Temperatura", temperaturaFinal);
          ultima_temperatura_ar_publicada = temperaturaFinal;

          Serial.println("Publicado: Temperatura do Ar");
        } else {
          Serial.println("Não publicado: Temperatura do Ar");
        }
      }

      avaliarRegraIrrigacaoTemperatura(temperaturaFinal);
    }

    sensor_etapa = 2;
    Serial.println("--------------------------------");
    return;
  }

  // ==================================================
  // ETAPA 2 -> UMIDADE DO AR
  // ==================================================
  if (sensor_etapa == 2) {
    Serial.println("Etapa 2/2 -> Umidade do Ar usando última leitura do DHT22");

    if (!ultima_leitura_dht_valida || isnan(ultima_umidade_ar_dht)) {
      Serial.println("Sem leitura válida recente do DHT22. Umidade do Ar não publicada.");
    } else {
      Serial.print("Umidade do Ar em cache: ");
      Serial.print(ultima_umidade_ar_dht);
      Serial.println(" %");

      if (my_horta) {
        if (devePublicarValor("Umidade do Ar", ultima_umidade_ar_dht, ultima_umidade_ar_publicada, DELTA_UMIDADE_AR)) {
          my_horta->updateAndReportParam("Umidade do Ar", ultima_umidade_ar_dht);
          ultima_umidade_ar_publicada = ultima_umidade_ar_dht;

          Serial.println("Publicado: Umidade do Ar");
        } else {
          Serial.println("Não publicado: Umidade do Ar");
        }
      }
    }

    sensor_etapa = 0;
    Serial.println("Ciclo completo dos sensores: Solo -> Temperatura -> Umidade do Ar");
    Serial.println("--------------------------------");
    return;
  }

  sensor_etapa = 0;
}
