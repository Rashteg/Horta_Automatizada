// 2026-04-21 20:35 America/Sao_Paulo - RainMaker com BLE/SoftAP + 3 cliques + LED + DHT22 + umidade do solo + gráficos + atualização a cada 20 s

#include <Arduino.h>
#include "RMaker.h"
#include "WiFi.h"
#include "WiFiProv.h"
#include <Preferences.h>
#include "DHT.h"

// ==============================
// CONFIGURAÇÕES DO DISPOSITIVO
// ==============================
const char *service_name = "PROV_C3_01";
const char *pop = "12345678";
const char *service_key = "12345678";  // senha do SoftAP temporário

// ESP32-C3 SuperMini
static const int gpio_led = 8;      // LED onboard (normalmente invertido)
static const int gpio_button = 9;   // botão BOOT (ativo em LOW)

// ==============================
// ESTADO DO DISPOSITIVO
// ==============================
bool switch_state = true;

// Device único RainMaker
static Device *my_horta = NULL;

// NVS / Preferences
Preferences prefs;
const char *PREF_NS = "dasrum_rm";
const char *PREF_SOFTAP = "softap_once";

// ==============================
// DHT22 - TEMPERATURA E UMIDADE DO AR
// ==============================
static const int DHTPIN = 4;
#define DHTTYPE DHT22
DHT dht(DHTPIN, DHTTYPE);

// Compensação manual da temperatura
float valorParaSubtrair = 0.0f;

// ==============================
// SENSOR DE UMIDADE DO SOLO
// ==============================
static const int PINO_SENSOR_SOLO = 1;  // GPIO1 analógico
static const int valorSeco = 3370;
static const int valorMolhado = 1520;

// ==============================
// INTERVALOS DE LEITURA / ENVIO
// ==============================
static unsigned long last_sensor_read_ms = 0;
static const unsigned long SENSOR_READ_INTERVAL_MS = 20000; // 20 s

// ==============================
// MODOS DE PROVISIONAMENTO
// ==============================
enum ProvisionMode {
  PROV_MODE_BLE = 0,
  PROV_MODE_SOFTAP = 1
};

ProvisionMode currentProvMode = PROV_MODE_BLE;

// ==============================
// CONTROLE DO BOTÃO
// ==============================
static bool last_button_state = HIGH;
static unsigned long press_start_ms = 0;

// Controle dos cliques
static uint8_t click_count = 0;
static unsigned long sequence_start_ms = 0;
static unsigned long last_click_ms = 0;

// ==============================
// AJUSTES DE TEMPO
// ==============================
static const unsigned long DEBOUNCE_MS = 40;
static const unsigned long SHORT_CLICK_MAX_MS = 1200;
static const unsigned long CLICK_GAP_MAX_MS = 2000;
static const unsigned long CLICK_SEQUENCE_MAX_MS = 6000;
static const uint8_t REQUIRED_CLICKS = 3;

// ==============================
// FUNÇÕES AUXILIARES
// ==============================
void setOnboardLed(bool on) {
  // LED invertido no ESP32-C3 SuperMini
  digitalWrite(gpio_led, on ? LOW : HIGH);
}

const char *getProvModeName() {
  return (currentProvMode == PROV_MODE_SOFTAP) ? "SoftAP" : "BLE";
}

const char *getQrTransportName() {
  return (currentProvMode == PROV_MODE_SOFTAP) ? "softap" : "ble";
}

void saveSoftAPOnce(bool enable) {
  prefs.begin(PREF_NS, false);
  prefs.putBool(PREF_SOFTAP, enable);
  prefs.end();
}

bool loadSoftAPOnce() {
  prefs.begin(PREF_NS, true);
  bool value = prefs.getBool(PREF_SOFTAP, false);
  prefs.end();
  return value;
}

void clearClickSequence() {
  click_count = 0;
  sequence_start_ms = 0;
  last_click_ms = 0;
}

void blinkOnceAcceptedClick() {
  bool previous_state = switch_state;
  setOnboardLed(!previous_state);
  delay(70);
  setOnboardLed(previous_state);
}

void blinkFeedback(uint8_t times, unsigned long onMs, unsigned long offMs) {
  bool previous_state = switch_state;

  for (uint8_t i = 0; i < times; i++) {
    setOnboardLed(true);
    delay(onMs);
    setOnboardLed(false);
    delay(offMs);
  }

  setOnboardLed(previous_state);
}

void rebootToSoftAPRecovery() {
  Serial.printf("Sequencia valida: %u cliques detectados.\n", REQUIRED_CLICKS);
  Serial.println("Entrando em modo temporario SoftAP...");
  blinkFeedback(3, 120, 120);
  saveSoftAPOnce(true);
  delay(300);
  ESP.restart();
}

void registerShortClick(unsigned long now) {
  if (click_count == 0) {
    click_count = 1;
    sequence_start_ms = now;
    last_click_ms = now;
  } else {
    if ((now - last_click_ms) > CLICK_GAP_MAX_MS ||
        (now - sequence_start_ms) > CLICK_SEQUENCE_MAX_MS) {
      click_count = 1;
      sequence_start_ms = now;
      last_click_ms = now;
    } else {
      click_count++;
      last_click_ms = now;
    }
  }

  blinkOnceAcceptedClick();
  Serial.printf("Clique curto %u/%u detectado.\n", click_count, REQUIRED_CLICKS);

  if (click_count >= REQUIRED_CLICKS && (now - sequence_start_ms) <= CLICK_SEQUENCE_MAX_MS) {
    clearClickSequence();
    rebootToSoftAPRecovery();
  }
}

// ==============================
// SENSOR DE SOLO
// ==============================
int lerMediaSensorSolo(int amostras) {
  long soma = 0;
  for (int i = 0; i < amostras; i++) {
    soma += analogRead(PINO_SENSOR_SOLO);
    delay(10);
  }
  return soma / amostras;
}

int calcularUmidadePercentual(int valorLido, int seco, int molhado) {
  int percentual = map(valorLido, seco, molhado, 0, 100);
  if (percentual < 0) percentual = 0;
  if (percentual > 100) percentual = 100;
  return percentual;
}

const char* faixaUmidadeBruta(int valorBruto) {
  if (valorBruto <= 3370 && valorBruto >= 2908) return "MUITO SECO";
  if (valorBruto <= 2907 && valorBruto >= 2445) return "SECO";
  if (valorBruto <= 2444 && valorBruto >= 1983) return "UMIDADE MEDIA";
  if (valorBruto <= 1982 && valorBruto >= 1520) return "MUITO UMIDO";

  if (valorBruto > 3370) return "ACIMA DA FAIXA CALIBRADA";
  if (valorBruto < 1520) return "ABAIXO DA FAIXA CALIBRADA";

  return "FAIXA INDEFINIDA";
}

// ==============================
// EVENTOS DO PROVISIONAMENTO
// ==============================
void sysProvEvent(arduino_event_t *sys_event) {
  switch (sys_event->event_id) {
    case ARDUINO_EVENT_PROV_START:
      if (currentProvMode == PROV_MODE_BLE) {
        Serial.printf("\nProvisioning Started with name \"%s\" and PoP \"%s\" on BLE\n", service_name, pop);
      } else {
        Serial.printf("\nProvisioning Started with name \"%s\", key \"%s\" and PoP \"%s\" on SoftAP\n",
                      service_name, service_key, pop);
      }

      Serial.println("Scan this QR code from the provisioning application for Provisioning.");
      WiFiProv.printQR(service_name, pop, getQrTransportName());

      Serial.println();
      Serial.println("=== DADOS DE RECUPERACAO ===");
      Serial.printf("Modo atual: %s\n", getProvModeName());
      Serial.printf("Nome do dispositivo: %s\n", service_name);
      Serial.printf("POP: %s\n", pop);
      if (currentProvMode == PROV_MODE_SOFTAP) {
        Serial.printf("Senha do SoftAP: %s\n", service_key);
      }
      Serial.println("============================");
      break;

    case ARDUINO_EVENT_PROV_INIT:
      Serial.printf("Inicializando provisionamento %s...\n", getProvModeName());
      WiFiProv.disableAutoStop(10000);
      break;

    case ARDUINO_EVENT_PROV_CRED_RECV:
      Serial.println("Recebendo credenciais de Wi-Fi...");
      break;

    case ARDUINO_EVENT_PROV_CRED_SUCCESS:
      Serial.println("Credenciais recebidas com sucesso.");
      WiFiProv.endProvision();

      if (currentProvMode == PROV_MODE_SOFTAP) {
        Serial.println("Provisionamento em SoftAP concluido.");
        Serial.println("No proximo boot o dispositivo voltara ao modo normal BLE.");
        saveSoftAPOnce(false);
      }
      break;

    case ARDUINO_EVENT_WIFI_STA_CONNECTED:
      Serial.println("Conectado ao Wi-Fi.");
      break;

    case ARDUINO_EVENT_WIFI_STA_GOT_IP:
      Serial.print("IP: ");
      Serial.println(WiFi.localIP());
      break;

    case ARDUINO_EVENT_WIFI_STA_DISCONNECTED:
      Serial.println("Wi-Fi desconectado.");
      break;

    default:
      break;
  }
}

// ==============================
// CALLBACK DO DEVICE NO APP
// ==============================
void write_callback(Device *device, Param *param, const param_val_t val,
                    void *priv_data, write_ctx_t *ctx) {
  const char *param_name = param->getParamName();

  if (strcmp(param_name, "Power") == 0) {
    switch_state = val.val.b;
    setOnboardLed(switch_state);

    Serial.printf("App enviou Power: %s\n", switch_state ? "LIGADO" : "DESLIGADO");

    param->updateAndReport(val);
  }
}

// ==============================
// ESCOLHA DO MODO DE BOOT
// ==============================
void selectProvisionModeAtBoot() {
  bool softap_once = loadSoftAPOnce();

  if (softap_once) {
    currentProvMode = PROV_MODE_SOFTAP;
    Serial.println("Modo temporario SoftAP carregado da memoria.");
  } else {
    currentProvMode = PROV_MODE_BLE;
    Serial.println("Modo normal BLE carregado.");
  }
}

// ==============================
// INICIA O PROVISIONAMENTO
// ==============================
void startProvisioning() {
  WiFi.onEvent(sysProvEvent);

  if (currentProvMode == PROV_MODE_BLE) {
    WiFiProv.initProvision(NETWORK_PROV_SCHEME_BLE,
                           NETWORK_PROV_SCHEME_HANDLER_FREE_BTDM);

    RMaker.start();

    WiFiProv.beginProvision(NETWORK_PROV_SCHEME_BLE,
                            NETWORK_PROV_SCHEME_HANDLER_FREE_BTDM,
                            NETWORK_PROV_SECURITY_1,
                            pop,
                            service_name);
  } else {
    WiFiProv.initProvision(NETWORK_PROV_SCHEME_SOFTAP,
                           NETWORK_PROV_SCHEME_HANDLER_NONE);

    RMaker.start();

    WiFiProv.beginProvision(NETWORK_PROV_SCHEME_SOFTAP,
                            NETWORK_PROV_SCHEME_HANDLER_NONE,
                            NETWORK_PROV_SECURITY_1,
                            pop,
                            service_name,
                            service_key);
  }
}

// ==============================
// BOTÃO BOOT
// ==============================
// clique curto = conta para recovery
// >3 s = Reset Wi-Fi
// >10 s = Factory Reset
// 3 cliques curtos em até 6 s = próximo boot em SoftAP temporário
void handleBootButton() {
  bool current_state = digitalRead(gpio_button);
  unsigned long now = millis();

  if (last_button_state == HIGH && current_state == LOW) {
    press_start_ms = now;
  }

  if (last_button_state == LOW && current_state == HIGH) {
    unsigned long press_time = now - press_start_ms;

    if (press_time < DEBOUNCE_MS) {
      last_button_state = current_state;
      return;
    }

    if (press_time > 10000) {
      Serial.println("Botao: Factory Reset.");
      clearClickSequence();
      RMakerFactoryReset(2);
    } else if (press_time > 3000) {
      Serial.println("Botao: Reset Wi-Fi.");
      clearClickSequence();
      RMakerWiFiReset(2);
    } else if (press_time <= SHORT_CLICK_MAX_MS) {
      registerShortClick(now);
    } else {
      Serial.println("Pressionamento ignorado (muito longo para clique curto e menor que 3 s).");
      clearClickSequence();
    }
  }

  if (click_count > 0) {
    if ((now - last_click_ms) > CLICK_GAP_MAX_MS ||
        (now - sequence_start_ms) > CLICK_SEQUENCE_MAX_MS) {
      clearClickSequence();
    }
  }

  last_button_state = current_state;
}

// ==============================
// LEITURA E PUBLICAÇÃO DOS SENSORES
// ==============================
void handleSensors() {
  unsigned long now = millis();

  if ((now - last_sensor_read_ms) < SENSOR_READ_INTERVAL_MS) {
    return;
  }

  last_sensor_read_ms = now;

  // ---------- DHT22 ----------
  float umidadeAr = dht.readHumidity();
  float temperaturaBruta = dht.readTemperature();

  if (isnan(umidadeAr) || isnan(temperaturaBruta)) {
    Serial.println("Falha ao ler o sensor DHT22! Verifique as conexoes.");
  } else {
    float temperaturaFinal = temperaturaBruta - valorParaSubtrair;

    Serial.print("Temp. Original: ");
    Serial.print(temperaturaBruta);
    Serial.print(" *C | Temp. Compensada: ");
    Serial.print(temperaturaFinal);
    Serial.print(" *C | Umidade do Ar: ");
    Serial.print(umidadeAr);
    Serial.println(" %");

    if (my_horta) {
      my_horta->updateAndReportParam("Temperatura", temperaturaFinal);
      delay(120);
      my_horta->updateAndReportParam("Umidade do Ar", umidadeAr);
      delay(120);
    }
  }

  // ---------- Sensor de Solo ----------
  int valorBrutoSolo = lerMediaSensorSolo(10);
  int umidadeSolo = calcularUmidadePercentual(valorBrutoSolo, valorSeco, valorMolhado);
  const char* faixaSolo = faixaUmidadeBruta(valorBrutoSolo);

  Serial.print("Solo -> Valor bruto: ");
  Serial.print(valorBrutoSolo);
  Serial.print(" | Umidade estimada: ");
  Serial.print(umidadeSolo);
  Serial.print("% | Faixa: ");
  Serial.println(faixaSolo);

  if (my_horta) {
    my_horta->updateAndReportParam("Umidade do Solo", (float)umidadeSolo);
  }

  Serial.println("--------------------------------");
}

// ==============================
// SETUP
// ==============================
void setup() {
  Serial.begin(115200);
  delay(1500);

  Serial.println();
  Serial.println("Bootando ESP RainMaker...");

  pinMode(gpio_led, OUTPUT);
  setOnboardLed(switch_state);

  pinMode(gpio_button, INPUT_PULLUP);
  last_button_state = digitalRead(gpio_button);

  analogReadResolution(12);
  dht.begin();

  Serial.println("Sensor de umidade do solo no ESP32-C3");
  Serial.println("Faixas brutas:");
  Serial.println("3370 ate 2908 -> MUITO SECO");
  Serial.println("2907 ate 2445 -> SECO");
  Serial.println("2444 ate 1983 -> UMIDADE MEDIA");
  Serial.println("1982 ate 1520 -> MUITO UMIDO");
  Serial.println("--------------------------------");

  selectProvisionModeAtBoot();

  Node my_node = RMaker.initNode("ESP32-C3 Inteligente");

  my_horta = new Device("Horta Inteligente", NULL, NULL);
  if (!my_horta) {
    Serial.println("Erro ao criar device Horta Inteligente.");
    return;
  }

  my_horta->addPowerParam(switch_state, "Power");

  Param temperature_param("Temperatura", "esp.param.temperature", value(0.0f),
                          PROP_FLAG_READ | PROP_FLAG_TIME_SERIES);
  my_horta->addParam(temperature_param);

  Param air_humidity_param("Umidade do Ar", "esp.param.humidity", value(0.0f),
                           PROP_FLAG_READ | PROP_FLAG_TIME_SERIES);
  my_horta->addParam(air_humidity_param);

  Param soil_humidity_param("Umidade do Solo", "esp.param.humidity", value(0.0f),
                            PROP_FLAG_READ | PROP_FLAG_TIME_SERIES);
  soil_humidity_param.addBounds(value(0.0f), value(100.0f), value(1.0f));
  my_horta->addParam(soil_humidity_param);

  my_horta->assignPrimaryParam(my_horta->getParamByName("Power"));
  my_horta->addCb(write_callback);

  my_node.addDevice(*my_horta);

  startProvisioning();

  Serial.printf("RainMaker iniciado em modo %s.\n", getProvModeName());
  Serial.println("Modo normal: BLE + QR.");
  Serial.println("Recuperacao sem QR: 3 cliques curtos no BOOT em ate 6 segundos.");
  Serial.println("O ESP reinicia em SoftAP temporario.");
  Serial.println("Apos provisionar com sucesso, o proximo boot volta ao BLE normal.");
  Serial.println("Botao BOOT: >3 s = Reset Wi-Fi | >10 s = Factory Reset");
  Serial.println("DHT22 ativo no GPIO4.");
  Serial.println("Sensor de solo ativo no GPIO1.");
  Serial.println("Atualizacao cloud/app a cada 20 segundos.");
  Serial.println("Graficos habilitados para Temperatura, Umidade do Ar e Umidade do Solo.");
}

// ==============================
// LOOP
// ==============================
void loop() {
  handleBootButton();
  handleSensors();
  delay(10);
}
